SELECT
    mp.Pack_Name,
    CONVERT(VARCHAR(5),
        DATEADD(HOUR, DATEDIFF(HOUR, 0, mch.completion_time), 0), 108)
    + ' - ' +
    CONVERT(VARCHAR(5),
        DATEADD(MINUTE, 59,
            DATEADD(HOUR, DATEDIFF(HOUR, 0, mch.completion_time), 0)
        ), 108) AS HourRange,
    COUNT(*) AS PackCount
FROM [taco_treceability_station_status_history].[dbo].[pack_completion_14_07_2026] mch
LEFT JOIN [taco_treceability_master].[taco_treceability].[master_pack] mp
    ON mch.pack_id = mp.Pack_ID
WHERE mch.station_name = 'Pack_Assembly'
GROUP BY
    mp.Pack_Name,
    DATEDIFF(HOUR, 0, mch.completion_time)
ORDER BY
    DATEDIFF(HOUR, 0, mch.completion_time),
    mp.Pack_Name;
